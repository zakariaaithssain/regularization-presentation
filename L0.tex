\section{L0 regularization}

\begin{frame}{$L_{0}$ regularization: definition}
\textbf{$L_{0}$ regularization} adds a penalty equal to the count of non‑zero weights (vector's sparsity) in the model:
\[
\norm{\omega}_0 = card(\{i \;/\; \omega_i \neq 0\}).
\]

\textbf{Penalized objective:}

\[
\min_{\omega \in \R^p} \; \mathcal{L}(\omega) + \lambda \, \norm{\omega}_0
\]
\begin{itemize}
  \item p: the number of weights
  \item $\mathcal{L}(\omega)$: loss function
  \item $\lambda>0$ regularization strength
\end{itemize}

\vspace{0.5em}
\alert{Interpretation:} \textbf{“minimize the loss function while using as few features as possible.”}
\end{frame}

\begin{frame}{Intuition: best weights subset selection}
\begin{itemize}
  \item $L_{0}$ is the most \textbf{direct and ideal} sparsity control.
  \item Often yields \textbf{very sparse} and \textbf{highly interpretable} models.
\end{itemize}
\end{frame}



\begin{frame}{$L_{0}$ regularization algorithm}
  \begin{itemize}
    \item choose an initial subset of weights : $best_S = S_{0}$\\
    \item for each subset S of the p weights:
  \begin{enumerate}
    \item Fix weights outside S to zero
    \item Optimize the weights in S to minimize the loss function
    \item Compute the objective function $Obj(\omega_{S})$
    \item if $Obj(\omega_{S}) \leq Obj(\omega_{best_S})$ : set $best_S$ to S
  \end{enumerate}
  \end{itemize}
  
    

\end{frame}


\begin{frame}{Why L0 is not Practical}
\begin{itemize}
  \item \textbf{Combinatorial problem in discrete space}: 
  \begin{itemize}
    \item for each weight, whether use it or prune it
    \item NP-Hard
    \item $2^{p}$ possiblities (far more than atoms in the universe for only 1million params)
  \end{itemize}
  \item \textbf{Non-differentiability}: Can't use Gradient based optimization (required by DL).
\end{itemize}
\end{frame}
